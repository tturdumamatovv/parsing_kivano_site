# запуск проекта

## Инструкция

## Работа с окружением
## Создать окружение
'''
python3 -m venv <название окружения>
'''

## Активация окружения
'''
. venv/bin/activate
'''
## или 
'''
source venv/bin/activate
'''
## Импорт библиотек и модулей
'''
pip install -a requirements.txt
'''
## Запуск процесса
'''
python3 main.py
'''
